import java.util.Arrays; // for Arrays.toString()

public class BubbleSort {

  private int[] arr;
  
  BubbleSort(int[] arr) {
    this.arr = arr;
  }

  public void sort() {
    for(int i = 0; i < arr.length; i++) { // loop through n amount of times
      for(int j = 0; j < arr.length; j++) { // loop through another n amount of times
        int temp; // for when you change arr[j]
        if(arr[j] > arr[j+1] {
          temp = arr[j];
          arr[j] = arr[j+1];
          arr[j+1] = temp;
        }
      }
    }
  }

  public String toString() {
    return "\n" + Arrays.toString(arr);
  }

}

  private int[] arr;
  
  BubbleSort(int[] arr) {
    this.arr = arr;
  }

  public void sort() {
    for(int i = 1; i < arr.length/2; i++) { // only go through half the length times
      for(int j = 0; j < arr.length - i; j++) { // end will be sorted so subtract i
        int temp; // for when you change arr[j]
        if(arr[j] > arr[j+1] {
          temp = arr[j];
          arr[j] = arr[j+1];
          arr[j+1] = temp;
        }
      }
    }
  }

  public String toString() {
    return "\n" + Arrays.toString(arr);
  }

}
